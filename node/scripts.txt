  `RecId` int NOT NULL AUTO_INCREMENT,

  `Email` varchar(45) DEFAULT NULL,
  `Password` varchar(45) DEFAULT NULL,
  `OTP` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`RecId`),
  UNIQUE KEY `Email_UNIQUE` (`Email`)
